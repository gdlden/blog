```go
import (
    "context"
    "github.com/go-kratos/kratos/v2/middleware/auth/jwt"
    jwtv5 "github.com/golang-jwt/jwt/v5"
)

func withUser(ctx context.Context, userId string) context.Context {
    return jwt.NewContext(ctx, jwtv5.MapClaims{
        "userId": userId,
    })
}
```
